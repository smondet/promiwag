module Env: Promiwag_std.Environment

type ('a, 'b) environment =
| Base of ('a * 'b) list
| Pushed of ('a * 'b) list * ('a, 'b) environment 
type ('a, 'b) t = ('a, 'b) environment 
val empty : ('a, 'b) environment
val of_list : ('a * 'b) list -> ('a, 'b) environment
val push : ('a, 'b) environment ->
       ('a, 'b) environment
val pop : ('a, 'b) environment ->
       ('a, 'b) environment
val add : ('a, 'b) environment ->
       'a -> 'b -> ('a, 'b) environment
val find_opt : ?cmp:('a -> 'a -> bool) ->
       ('a, 'b) environment -> 'a -> 'b option
val iter : f:('a * 'b -> unit) -> ('a, 'b) environment -> unit
val metamap : map:('a * 'b -> 'c) ->
       reduce:('c list -> 'd) ->
       ('a, 'b) environment -> 'd list
